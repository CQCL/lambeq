[docs]    def tokenise_sentence(self, sentence: str) -> list[str]:
        """Tokenise a sentence.

        Parameters
        ----------
        sentence : str
            An untokenised sentence.

        Returns
        -------
        list of str
            A tokenised sentence given as a list of tokens - strings.

        """
        return self.tokenise_sentences([sentence])[0]
